class Solution {
public:
    int kthSmallest(vector<vector<int>> mat, int k) {
        int n = mat.size();
    
    if(k > n*n)
    return -1;
    
    
    if(k == 1)
    return mat[0][0];
    
        vector <int> arr;
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
        arr.push_back(mat[i][j]);
    
    
    sort(arr.begin(),arr.end());
    
    return arr[k-1];
        
    }
};
